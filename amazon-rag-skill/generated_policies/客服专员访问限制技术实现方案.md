class CustomerServiceAccessBlock {
    constructor() {
        this.blockedUrls = [
            'sellercentral.amazon.com/sw/AccountInfo',
            'sellercentral.amazon.com/sw/AccountInfo/',
            'sellercentral.amazon.com/sw/AccountInfo/*'
        ];
        this.userRole = this.getCurrentUserRole();
        this.init();
    }

    init() {
        if (this.isCustomerServiceRole()) {
            this.setupUrlBlocking();
            this.setupDOMObserver();
            this.setupNavigationBlocking();
        }
    }

    isCustomerServiceRole() {
        const customerServiceRoles = [
            '客服专员', '客服助理', '初级客服', 
            '售后专员', '客户支持', '在线客服', '电话客服'
        ];
        return customerServiceRoles.includes(this.userRole);
    }

    setupUrlBlocking() {
        // 拦截页面导航
        window.addEventListener('beforeunload', (e) => {
            const targetUrl = e.target.activeElement?.href || window.location.href;
            if (this.isBlockedUrl(targetUrl)) {
                e.preventDefault();
                this.showBlockScreen();
                return false;
            }
        });

        // 拦截所有链接点击
        document.addEventListener('click', (e) => {
            const link = e.target.closest('a');
            if (link && this.isBlockedUrl(link.href)) {
                e.preventDefault();
                e.stopPropagation();
                this.showBlockScreen();
            }
        }, true);
    }

    isBlockedUrl(url) {
        if (!url) return false;
        return this.blockedUrls.some(blocked => 
            url.toLowerCase().includes(blocked.toLowerCase())
        );
    }
